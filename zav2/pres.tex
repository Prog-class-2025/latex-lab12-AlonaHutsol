% 3. СПИСКИ (Поступова поява)
\section{Загальні відомості}
\begin{frame}{Що таке квадратне рівняння?}
    \begin{block}{Означення}
        Квадратним рівнянням називається рівняння вигляду
        \[
        ax^2 + bx + c = 0,
        \]
        де $a$, $b$, $c$ --- дійсні числа, причому $a \neq 0$.
    \end{block}

    \begin{itemize}
        \item<2-> $a$ --- коефіцієнт при $x^2$;
        \item<3-> $b$ --- коефіцієнт при $x$;
        \item<4-> $c$ --- вільний член.
    \end{itemize}
\end{frame}

% 4. МАТЕМАТИКА ТА ЯСКРАВІ РАМКИ
\section{Теоретичні основи}
\begin{frame}{Дискримінант квадратного рівняння}
    \begin{block}{Означення}
        Для квадратного рівняння
        \[
        ax^2 + bx + c = 0
        \]
        дискримінант обчислюється за формулою
        \[
        D = b^2 - 4ac.
        \]
    \end{block}

    \begin{exampleblock}{Кількість коренів}
        \begin{itemize}
            \item Якщо $D > 0$, рівняння має два різні корені.
            \item Якщо $D = 0$, рівняння має один корінь.
            \item Якщо $D < 0$, дійсних коренів немає.
        \end{itemize}
    \end{exampleblock}
\end{frame}
% 5. ГРАФІКА ТА КОЛОНКИ
\begin{frame}{Формула коренів квадратного рівняння}
    \begin{block}{Теорема}
        Якщо дискримінант квадратного рівняння
        \[
        ax^2 + bx + c = 0
        \]
        невід’ємний, то його корені обчислюються за формулою
        \[
        x_{1,2} = \frac{-b \pm \sqrt{D}}{2a}.
        \]
    \end{block}

    \begin{alertblock}{Увага!}
        Якщо $D < 0$, то в множині дійсних чисел квадратне рівняння коренів не має.
    \end{alertblock}
\end{frame}

\section{Приклад розв’язування}
\begin{frame}{Розв’язування прикладу}
    Розв’яжемо рівняння
    \[
    x^2 - 5x + 6 = 0.
    \]

    \begin{exampleblock}{Крок 1. Знайдемо дискримінант}
        \[
        D = (-5)^2 - 4 \cdot 1 \cdot 6 = 25 - 24 = 1.
        \]
    \end{exampleblock}

    \begin{exampleblock}{Крок 2. Обчислимо корені}
        \[
        x_1 = \frac{5 + \sqrt{1}}{2} = 3,
        \qquad
        x_2 = \frac{5 - \sqrt{1}}{2} = 2.
        \]
    \end{exampleblock}
\end{frame}

\begin{frame}{Графік квадратної функції}
    \begin{columns}

            \[
            y = ax^2 + bx + c.

            \begin{itemize}
                \item Якщо $a>0$, вітки параболи напрямлені вгору.
                \item Якщо $a<0$, вітки параболи напрямлені вниз.
                \item Корені рівняння --- це точки перетину графіка з віссю $Ox$.

\section{Практичне значення}
\begin{frame}{Де застосовуються квадратні рівняння?}
    \begin{enumerate}
        \item Розв’язування задач з геометрії.
        \item Обчислення траєкторій у фізиці.
        \item Економічні розрахунки.
        \item Побудова графіків функцій.
    \end{enumerate}

    \vspace{0.4cm}

    \begin{block}{Приклад}
        Квадратні рівняння часто виникають у задачах на площу, швидкість, рух та моделювання реальних процесів.
    \end{block}
\end{frame}
